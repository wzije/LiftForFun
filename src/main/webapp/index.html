# Architecture must be like this, so you must create file or directory manually on first create project. 
    - project root directory
  | build.sbt
  - project/
    | plugins.sbt
  - src/
    - main/
      - scala/
        - bootstrap/
          | Boot.scala
        - org/
          - yourorganization/
            - liftfromscratch/
              | =>your Scala code goes here
    - webapp/
    | index.html
    | =>any other web resources - images, HTML, JavaScript, etc - go here
    - WEB-INF/
    | web.xml
    - test/
    - scala/
    - org/
    - yourorganization/
    - liftfromscratch/
    | =>your tests go here
